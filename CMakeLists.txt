cmake_minimum_required(VERSION 3.14)
project(LogosPackageManagerModulePlugin LANGUAGES CXX)

# Include the Logos Module CMake helper
if(DEFINED ENV{LOGOS_MODULE_BUILDER_ROOT})
    include($ENV{LOGOS_MODULE_BUILDER_ROOT}/cmake/LogosModule.cmake)
else()
    message(FATAL_ERROR "LogosModule.cmake not found. Set LOGOS_MODULE_BUILDER_ROOT.")
endif()

# Define the module (uses LogosProviderBase / LOGOS_METHOD API)
logos_module(
    NAME package_manager
    SOURCES
        src/package_manager_impl.h
        src/package_manager_impl.cpp
        src/package_manager_loader.h
    EXTERNAL_LIBS
        package_manager_lib
        lgx
)

# Provider dispatch file (generated by logos-cpp-generator --provider-header
# during preConfigure). Added separately without GENERATED property to avoid
# an AUTOMOC dependency cycle with ninja.
set(_DISPATCH "${CMAKE_CURRENT_SOURCE_DIR}/generated_code/logos_provider_dispatch.cpp")
if(EXISTS "${_DISPATCH}")
    target_sources(package_manager_module_plugin PRIVATE "${_DISPATCH}")
endif()
